001    package nl.jworks.grails.plugin.fitnesse;
002    
003    import org.codehaus.groovy.grails.commons.AbstractInjectableGrailsClass;
004    
005    /**
006     * @author Erik Pragt
007     */
008    public class DefaultGrailsFitnesseFixtureClass extends AbstractInjectableGrailsClass implements GrailsFitnesseFixtureClass {
009    
010        private final static String QUERY_FIXTURE = "queryFixture";
011        private final static String MAPPING = "mapping";
012    
013        private boolean isQueryFixture;
014        private Object mapping;
015    
016        public DefaultGrailsFitnesseFixtureClass(Class clazz) {
017            super(clazz, FitnesseFixtureArtefactHandler.CLASSNAME_SUFFIX);
018    
019            Object tmpIsQueryFixture = getStaticPropertyValue(QUERY_FIXTURE, Boolean.class);
020            isQueryFixture = Boolean.TRUE.equals(tmpIsQueryFixture);
021    
022            mapping = getStaticPropertyValue(MAPPING, Object.class);
023        }
024    
025        public boolean getIsQueryFixture() {
026            return isQueryFixture;
027        }
028    
029        public Object getMapping() {
030            return mapping;
031        }
032    }
